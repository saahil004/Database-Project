CREATE TABLE customers (
	customer_id INT AUTO_INCREMENT,
	name VARCHAR(100) NOT NULL,
	address VARCHAR(100) NOT NULL,
	username VARCHAR(100) UNIQUE NOT NULL,
	email VARCHAR(100) UNIQUE NOT NULL,
	password VARCHAR(255) NOT NULL,
	contact VARCHAR(20) NOT NULL,
	PRIMARY KEY (customer_id)
);

CREATE TABLE menuitems (
	menu_item_id INT 
	AUTO_INCREMENT,
	name VARCHAR(100) NOT NULL,
	category_id INT NOT NULL,
	quantity INT NOT NULL,
	price DECIMAL(10,2) NOT NULL,
	PRIMARY KEY(menu_item_id),
	FOREIGN KEY (category_id) REFERENCES categories(category_id)
);

CREATE TABLE orders (
	order_id INT AUTO_INCREMENT,
	customer_id INT,
	order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	total_bill DECIMAL(10,2) NOT NULL,
	status VARCHAR(50) NOT NULL,

	PRIMARY KEY(order_id),
	FOREIGN KEY (customer_id) REFERENCES customers( customer_id)
);

CREATE TABLE deliveryguy (
	deliveryguy_id INT AUTO_INCREMENT,
	name VARCHAR(100) NOT NULL,
	username VARCHAR(100) UNIQUE NOT NULL,
	email VARCHAR(100) UNIQUE NOT NULL,
	password VARCHAR(255) NOT NULL,
	contact VARCHAR(20) NOT NULL,
	PRIMARY KEY (deliveryguy_id)
);




CREATE TABLE admins (
	admin_id INT AUTO_INCREMENT,
	name VARCHAR(100) NOT NULL,
	username VARCHAR(100) UNIQUE NOT NULL,
	email VARCHAR(100) UNIQUE NOT NULL,
	password VARCHAR(255) NOT NULL,
	PRIMARY KEY (admin_id)
);


CREATE TABLE orderitems (
	order_id INT,
	menu_item_id INT,
	quantity INT DEFAULT 1 NOT NULL CHECK (quantity > 0),
	PRIMARY KEY (order_id, menu_item_id),
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (menu_item_id) REFERENCES menuitems(menu_item_id) ON DELETE CASCADE ON UPDATE CASCADE
);


CREATE TABLE delivery (
	delivery_id INT AUTO_INCREMENT,
	order_id INT,
	deliveryguy_id INT,
	deliverystatus VARCHAR(50) NOT NULL,
	delivery_time DATETIME,
	PRIMARY KEY (delivery_id),
	FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (deliveryguy_id) REFERENCES deliveryguy(deliveryguy_id) ON DELETE CASCADE ON UPDATE CASCADE
);


CREATE TABLE categories (
	category_id INT AUTO_INCREMENT,
	name VARCHAR(40) UNIQUE NOT NULL,
	PRIMARY KEY(category_id)
);


Triggers:
- reduce stock after order
- prevent ordering out of stock items
- assign delivery guy after order is confirmed
DELIMITER $$

CREATE TRIGGER auto_assign_delivery
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    DECLARE selected_guy INT;

    -- Run only when status changes to 'preparing'
    IF NEW.status = 'preparing' AND OLD.status != 'preparing' THEN

        -- Find delivery guy with minimum active orders
        SELECT dg.deliveryguy_id
        INTO selected_guy
        FROM deliveryguy dg
        LEFT JOIN delivery d 
            ON dg.deliveryguy_id = d.deliveryguy_id 
            AND d.deliverystatus != 'delivered'
        GROUP BY dg.deliveryguy_id
        ORDER BY COUNT(d.delivery_id) ASC
        LIMIT 1;

        -- Assign delivery
        INSERT INTO delivery (order_id, deliveryguy_id, deliverystatus)
        VALUES (NEW.order_id, selected_guy, 'assigned');

    END IF;

END$$

DELIMITER ;

CREATE TRIGGER trg_check_stock
BEFORE INSERT ON orderitems
FOR EACH ROW
BEGIN
    DECLARE available_stock INT;

    SELECT quantity INTO available_stock
    FROM menuitems
    WHERE menu_item_id = NEW.menu_item_id;

    IF available_stock < NEW.quantity THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Not enough stock available';
    END IF;
END//

DELIMITER ;

CREATE TRIGGER trg_update_total
AFTER INSERT ON orderitems
FOR EACH ROW
BEGIN
    UPDATE orders
    SET total_bill = (
        SELECT SUM(oi.quantity * mi.price)
        FROM orderitems oi
        JOIN menuitems mi ON oi.menu_item_id = mi.menu_item_id
        WHERE oi.order_id = NEW.order_id
    )
    WHERE order_id = NEW.order_id;
END//

CREATE TRIGGER reduce_stock_on_confirmed
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    DECLARE done INT DEFAULT 0;
    DECLARE v_menu_item_id INT;
    DECLARE v_quantity INT;

    -- Cursor to loop through all items in the order
    DECLARE order_cursor CURSOR FOR
        SELECT menu_item_id, quantity
        FROM orderitems
        WHERE order_id = NEW.order_id;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

    -- Only proceed if status changed from pending to confirmed
    IF OLD.status = 'pending' AND NEW.status = 'preparing' THEN

        OPEN order_cursor;

        read_loop: LOOP
            FETCH order_cursor INTO v_menu_item_id, v_quantity;
            IF done THEN
                LEAVE read_loop;
            END IF;

            -- Check stock availability
            IF (SELECT quantity FROM menuitems WHERE menu_item_id = v_menu_item_id) < v_quantity THEN
                CLOSE order_cursor;
                SIGNAL SQLSTATE '45000'
                SET MESSAGE_TEXT = 'Insufficient stock for one or more items';
            END IF;

            -- Deduct stock
            UPDATE menuitems
            SET quantity = quantity - v_quantity
            WHERE menu_item_id = v_menu_item_id;

        END LOOP;

        CLOSE order_cursor;

    END IF;
END$$

DELIMITER ;
